export default async function handler(req,res){ try{ const { token, ...logins } = req.query; const loginParams = Object.values(logins) .map(l => `user_login=${l}`) .join("&"); const response = await fetch( `https://api.twitch.tv/helix/streams?${loginParams}`, { headers:{ Authorization:`Bearer ${token}`, "Client-Id": process.env.CLIENT_ID } } ); const data = await response.json(); res.status(200).json(data); }catch(err){ res.status(500).json({error:err.message}); } }
